115     @Override
116     public ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs) {
117         requireNonNull(serviceConfigs, "serviceConfigs");
118         final Map<Class<?>, Set<MethodInfo>> methodInfos = new HashMap<>();
119         final Map<Class<?>, String> serviceDescription = new HashMap<>();
120         serviceConfigs.forEach(sc -> {
121             final Optional<AnnotatedHttpService> service = sc.service().as(AnnotatedHttpService.class);
122             service.ifPresent(
123                     httpService -> {
124                         addMethodInfo(methodInfos, sc.virtualHost().hostnamePattern(), httpService);
125                         addServiceDescription(serviceDescription, httpService);
126                     });
127         });
128 
129         return generate(serviceDescription, methodInfos);
130     }
131 
132     private void addServiceDescription(Map<Class<?>, String> serviceDescription, AnnotatedHttpService service) {
133         final Class<?> clazz = service.object().getClass();
134         serviceDescription.computeIfAbsent(clazz, AnnotatedHttpServiceFactory::findDescription);
135     }
136 
137     private static void addMethodInfo(Map<Class<?>, Set<MethodInfo>> methodInfos,
138                                       String hostnamePattern, AnnotatedHttpService service) {
139         final PathMapping pathMapping = service.pathMapping();
140         final EndpointInfo endpoint = endpointInfo(pathMapping, hostnamePattern);
141         if (endpoint == null) {
142             return;
143         }
144 
145         final Method method = service.method();
146         final String name = method.getName();
147         final TypeSignature returnTypeSignature = toTypeSignature(method.getGenericReturnType());
148         final List<FieldInfo> fieldInfos = fieldInfos(service.annotatedValueResolvers());
149         final Class<?> clazz = service.object().getClass();
150         pathMapping.supportedMethods().forEach(
151                 httpMethod -> {
152                     final MethodInfo methodInfo = new MethodInfo(
153                             name, returnTypeSignature, fieldInfos, ImmutableList.of(), // Ignore exceptions.
154                             ImmutableList.of(endpoint), httpMethod, findDescription(method));
155                     methodInfos.computeIfAbsent(clazz, unused -> new HashSet<>()).add(methodInfo);
156                 });
157     }

159     @Nullable
160     @VisibleForTesting
161     static EndpointInfo endpointInfo(PathMapping pathMapping, String hostnamePattern) {
162         final String endpointPathMapping = endpointPathMapping(pathMapping);
163         if (isNullOrEmpty(endpointPathMapping)) {
164             return null;
165         }
166 
167         final EndpointInfoBuilder builder = new EndpointInfoBuilder(hostnamePattern, endpointPathMapping);
168         if (endpointPathMapping.startsWith(REGEX) && pathMapping.prefix().isPresent()) {
169             // PrefixAddingPathMapping
170             builder.regexPathPrefix(PREFIX + pathMapping.prefix().get());
171         }
172 
173         builder.availableMimeTypes(availableMimeTypes(pathMapping));
174         return builder.build();
175     }
176 
177     private static String endpointPathMapping(PathMapping pathMapping) {
178         final Optional<String> exactPath = pathMapping.exactPath();
179         if (exactPath.isPresent()) {
180             return EXACT + exactPath.get();
181         }
182 
183         final Optional<String> regex = pathMapping.regex();
184         if (regex.isPresent()) {
185             return REGEX + regex.get();
186         }
187 
188         final Optional<String> prefix = pathMapping.prefix();
189         if (prefix.isPresent()) {
190             return PREFIX + prefix.get();
191         }
192 
193         return getNormalizedTriePath(pathMapping);
194     }
195 
196     private static List<MediaType> availableMimeTypes(PathMapping pathMapping) {
197         final Builder<MediaType> builder = ImmutableList.builder();
198         final List<MediaType> consumeTypes = pathMapping.consumeTypes();
199         builder.addAll(consumeTypes);
200         if (!consumeTypes.contains(MediaType.JSON_UTF_8)) {
201             builder.add(MediaType.JSON_UTF_8);
202         }
203         return builder.build();
204     }

206     @VisibleForTesting
207     static List<FieldInfo> fieldInfos(List<AnnotatedValueResolver> resolvers) {
208         final ImmutableList.Builder<FieldInfo> fieldInfoBuilder = ImmutableList.builder();
209         for (AnnotatedValueResolver resolver : resolvers) {
210             if (!(resolver.annotationType() == Param.class || resolver.annotationType() == Header.class)) {
211                 continue;
212             }
213             final TypeSignature signature;
214             if (resolver.hasContainer()) {
215                 final Class<?> containerType = resolver.containerType();
216                 assert containerType != null;
217                 final TypeSignature parameterTypeSignature = toTypeSignature(resolver.elementType());
218                 if (List.class.isAssignableFrom(containerType)) {
219                     signature = TypeSignature.ofList(parameterTypeSignature);
220                 } else if (Set.class.isAssignableFrom(containerType)) {
221                     signature = TypeSignature.ofSet(parameterTypeSignature);
222                 } else {
223                     // Only List and Set are supported for the containerType.
224                     continue;
225                 }
226             } else {
227                 signature = toTypeSignature(resolver.elementType());
228             }
229             final String name = resolver.httpElementName();
230             assert name != null;
231 
232             fieldInfoBuilder.add(
233                     new FieldInfo(name, location(resolver), resolver.shouldExist() ? FieldRequirement.REQUIRED
234                                                                                    : FieldRequirement.OPTIONAL,
235                                   signature, resolver.description()));
236         }
237         return fieldInfoBuilder.build();
238     }

240     @VisibleForTesting
241     static TypeSignature toTypeSignature(Type type) {
242         requireNonNull(type, "type");
243 
244         // The data types defined by the OpenAPI Specification:
245 
246         if (type == Void.class || type == void.class) {
247             return VOID;
248         } else if (type == Boolean.class || type == boolean.class) {
249             return BOOL;
250         } else if (type == Byte.class || type == byte.class) {
251             return INT8;
252         } else if (type == Short.class || type == short.class) {
253             return INT16;
254         } else if (type == Integer.class || type == int.class) {
255             return INT32;
256         } else if (type == Long.class || type == long.class) {
257             return INT64;
258         } else if (type == Float.class || type == float.class) {
259             return FLOAT;
260         } else if (type == Double.class || type == double.class) {
261             return DOUBLE;
262         } else if (type == String.class) {
263             return STRING;
264         } else if (type == byte[].class || type == Byte[].class) {
265             return BINARY;
266         }
267         // End of data types defined by the OpenAPI Specification.
268 
269         if (type instanceof ParameterizedType) {
270             final ParameterizedType parameterizedType = (ParameterizedType) type;
271             final Class<?> rawType = (Class<?>) parameterizedType.getRawType();
272             if (List.class.isAssignableFrom(rawType)) {
273                 return TypeSignature.ofList(toTypeSignature(parameterizedType.getActualTypeArguments()[0]));
274             }
275             if (Set.class.isAssignableFrom(rawType)) {
276                 return TypeSignature.ofSet(toTypeSignature(parameterizedType.getActualTypeArguments()[0]));
277             }
278 
279             if (Map.class.isAssignableFrom(rawType)) {
280                 final TypeSignature key = toTypeSignature(parameterizedType.getActualTypeArguments()[0]);
281                 final TypeSignature value = toTypeSignature(parameterizedType.getActualTypeArguments()[1]);
282                 return TypeSignature.ofMap(key, value);
283             }
284 
285             final List<TypeSignature> actualTypes = Stream.of(parameterizedType.getActualTypeArguments())
286                                                           .map(AnnotatedHttpDocServicePlugin::toTypeSignature)
287                                                           .collect(toImmutableList());
288             return TypeSignature.ofContainer(rawType.getSimpleName(), actualTypes);
289         }
290 
291         if (type instanceof WildcardType) {
292             // Create an unresolved type with an empty string so that the type name will be '?'.
293             return TypeSignature.ofUnresolved("");
294         }
295         if (type instanceof TypeVariable) {
296             return TypeSignature.ofBase(type.getTypeName());
297         }
298         if (type instanceof GenericArrayType) {
299             return TypeSignature.ofList(toTypeSignature(((GenericArrayType) type).getGenericComponentType()));
300         }
301 
302         if (!(type instanceof Class)) {
303             return TypeSignature.ofBase(type.getTypeName());
304         }
305 
306         final Class<?> clazz = (Class<?>) type;
307         if (clazz.isArray()) {
308             // If it's an array, return it as a list.
309             return TypeSignature.ofList(toTypeSignature(clazz.getComponentType()));
310         }
311 
312         return TypeSignature.ofBase(clazz.getSimpleName());
313     }

315     @Nullable
316     private static String location(AnnotatedValueResolver resolver) {
317         if (resolver.isPathVariable()) {
318             return PATH_PARAM;
319         }
320         if (resolver.annotationType() == Param.class) {
321             return QUERY_PARAM;
322         }
323         if (resolver.annotationType() == Header.class) {
324             return HEADER_PARAM;
325         }
326         return null;
327     }

329     @VisibleForTesting
330     static ServiceSpecification generate(Map<Class<?>, String> serviceDescription,
331                                          Map<Class<?>, Set<MethodInfo>> methodInfos) {
332         final Set<ServiceInfo> serviceInfos = methodInfos
333                 .entrySet().stream()
334                 .map(entry -> {
335                     final Class<?> service = entry.getKey();
336                     return new ServiceInfo(service.getName(), entry.getValue(),
337                                            serviceDescription.get(service));
338                 })
339                 .collect(toImmutableSet());
340 
341         return ServiceSpecification.generate(serviceInfos, AnnotatedHttpDocServicePlugin::newNamedTypeInfo);
342     }
343 
344     private static NamedTypeInfo newNamedTypeInfo(TypeSignature typeSignature) {
345         final Class<?> type = (Class<?>)
346                 typeSignature.namedTypeDescriptor()
347                              .orElseThrow(
348                                      () -> new IllegalArgumentException("cannot create a named type from: " +
349                                                                         typeSignature));
350         if (type.isEnum()) {
351             @SuppressWarnings("unchecked")
352             final Class<? extends Enum<?>> enumType = (Class<? extends Enum<?>>) type;
353             return new EnumInfo(enumType);
354         }
355 
356         return newStructInfo(type);
357     }
358 
359     private static StructInfo newStructInfo(Class<?> structClass) {
360         final String name = structClass.getName();
361 
362         final Field[] declaredFields = structClass.getDeclaredFields();
363         final List<FieldInfo> fields =
364                 Stream.of(declaredFields)
365                       .map(f -> new FieldInfo(f.getName(), FieldRequirement.DEFAULT,
366                                               toTypeSignature(f.getGenericType())))
367                       .collect(Collectors.toList());
368         return new StructInfo(name, fields);
369     }

376     @Override
377     public Optional<String> serializeExampleRequest(String serviceName, String methodName,
378                                                     Object exampleRequest) {
379         try {
380             return Optional.of(mapper.writerWithDefaultPrettyPrinter().writeValueAsString(exampleRequest));
381         } catch (JsonProcessingException e) {
382             // Ignore the exception and just return Optional.empty().
383         }
384         return Optional.empty();
385     }
